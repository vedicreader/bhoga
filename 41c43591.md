# bhoga — Implementation Specification

> **`bhoga`** (Sanskrit: *route, path, enjoyment of resources*) — a zero-server Python library
> that routes AI harnesses to the best available provider for a model, based on live subscription quota.
>
> Hand this document to Copilot + Claude Opus. Read §14 for ordered implementation steps.
> Confirm understanding before writing any code.

---

## 1. Problem

AI coding harnesses run against multiple providers that serve the **same underlying model**.
`claude-opus-4-20250514` is available via Claude Code Max, Anthropic API, and GitHub Copilot.
`gpt-4.1` is available via Codex CLI, OpenAI API, and GitHub Copilot.

No existing library:
- Tracks subscription quota state **across** providers for the same model
- Tells a harness *which provider to use right now* without a server or load balancer
- Gives first-class support for Hermes engine's `config.yaml` model-switching

## 2. Design Principles

### Coding Style: fastai / fastcore

```python
# ✅ succinct — one expression, annotated, no ceremony
def model_family(m: str) -> str:
    return next((f for p, f in _FAM.items() if m.lower().startswith(p)), "other")

# ❌ avoid
def model_family(model_id: str) -> str:
    for prefix, family in _FAM.items():
        if model_id.lower().startswith(prefix):
            return family
    return "other"
```

- `fastcore` utilities: `store_attr`, `ifnone`, `L`, `@threaded`
- Properties over getters; dataclasses over Pydantic
- `StrEnum` + `auto()` for all enums
- walrus operator (`:=`) for guard clauses
- No classes where a function suffices

### Dependencies

```toml
dependencies = ["fastcore>=1.7", "httpx>=0.27", "python-dateutil>=2.9"]
```

No LiteLLM. No server. No YAML file — defaults are embedded as a dict.

## 3. Architecture

```
Harness (Hermes / Claude Code / Codex / Cursor)
    │
    ├─ router.best_for("claude-opus-4")   → RouterRecommendation  [instant, in-memory]
    │
    └─ router.record_turn(provider, tokens, headers, status_code)  [instant, background write]

Background thread (fires on init + after each record_turn):
    init:  parse_cli_quota() → merge into _state
           fetch_models()    → populate provider×model entries
           load_state()      → restore consumed counts, reset if period elapsed
    turn:  calibrate_from_headers() → update calibrated_remaining
           save_state()             → write ~/.cache/bhoga/state.json
```

## 6. Complete Implementation — `bhoga.py`

```python
# bhoga.py — complete implementation (~250 lines)
from __future__ import annotations
import json, os, subprocess
from concurrent.futures import ThreadPoolExecutor
from dataclasses import dataclass, field
from datetime import datetime, timezone, timedelta
from enum import auto
from pathlib import Path
from typing import Callable, Literal

try:
    from enum import StrEnum
except ImportError:
    from strenum import StrEnum      # pip install strenum for Python 3.10

import httpx
from dateutil.relativedelta import relativedelta
from fastcore.basics import ifnone

# ── Defaults (embedded — no YAML file needed) ─────────────────────────────

_DEFAULTS = { ... }   # paste the full dict from §4 here

_FAM = {"claude": "anthropic", "gpt": "openai", "o1": "openai",
        "o3": "openai", "o4": "openai", "gemini": "google", "codex": "openai"}

STATE_PATH = Path.home() / ".cache" / "bhoga" / "state.json"

# ── Enums ─────────────────────────────────────────────────────────────────

class BillingCadence(StrEnum):
    HOURLY  = auto();  DAILY = auto();  WEEKLY = auto();  MONTHLY = auto()

class ProviderStatus(StrEnum):
    OK = auto();  LOW = auto();  THROTTLED = auto();  EXHAUSTED = auto();  UNKNOWN = auto()

# ── Dataclasses ───────────────────────────────────────────────────────────

@dataclass
class ProviderQuota:
    provider_id:  str
    model_id:     str
    plan:         str            = "unknown"
    ceiling:      int | None     = None          # tokens per cadence period
    cadence:      BillingCadence = BillingCadence.MONTHLY
    reset_at:     datetime | None = None
    consumed:     int            = 0             # tokens used this period (local tracking)
    n_requests:   int            = 0             # requests used this period
    calibrated:   int | None     = None          # authoritative remaining from header
    calibrated_at:datetime | None = None
    status:       ProviderStatus = ProviderStatus.UNKNOWN
    backoff_until:datetime | None = None
    priority:     int            = 99
    raw_cli:      str | None     = None          # kept for re-parse on next session

    @property
    def quota_pct(self) -> float:
        "Remaining [0,1]. Prefers header-calibrated over locally computed. -1 = unknown."
        if self.calibrated is not None and self.ceiling:
            return max(0.0, self.calibrated / self.ceiling)
        if self.ceiling: return max(0.0, 1.0 - self.consumed / self.ceiling)
        return -1.0

    def is_usable(self, thr=0.05) -> bool:
        now = datetime.now(timezone.utc)
        if self.status == ProviderStatus.THROTTLED and self.backoff_until and now < self.backoff_until:
            return False
        p = self.quota_pct
        return p < 0 or p > thr           # unknown quota = route optimistically

    def recompute(self, low=0.20) -> None:
        "Update status from current quota_pct + backoff. Call after any mutation."
        p, now = self.quota_pct, datetime.now(timezone.utc)
        if   self.backoff_until and now < self.backoff_until: self.status = ProviderStatus.THROTTLED
        elif 0 <= p <= 0.05:    self.status = ProviderStatus.EXHAUSTED
        elif 0 <= p <= low:     self.status = ProviderStatus.LOW
        elif p < 0:             self.status = ProviderStatus.UNKNOWN
        else:                   self.status = ProviderStatus.OK

    def reset_if_due(self) -> None:
        if self.reset_at and datetime.now(timezone.utc) > self.reset_at:
            self.consumed = self.n_requests = 0
            self.calibrated = None
            self.reset_at += {BillingCadence.HOURLY:  relativedelta(hours=1),
                              BillingCadence.DAILY:   relativedelta(days=1),
                              BillingCadence.WEEKLY:  relativedelta(weeks=1),
                              BillingCadence.MONTHLY: relativedelta(months=1)}[self.cadence]

@dataclass
class RouterRecommendation:
    "What best_for() returns. provider_id + model_id tell harness where to send the request."
    provider_id:   str
    model_id:      str
    quota_pct:     float
    status:        ProviderStatus
    reason:        str            # "highest_quota"|"priority_tiebreak"|"only_available"|"unknown_quota"|"all_exhausted"
    hermes_model:  str = ""       # ready-to-use Hermes model string: "anthropic/claude-opus-4-20250514"
    all_providers: dict[str, ProviderQuota] = field(default_factory=dict)

    @property
    def available(self) -> bool: return self.status != ProviderStatus.EXHAUSTED

    def __repr__(self):
        pct = f"{self.quota_pct:.0%}" if self.quota_pct >= 0 else "?"
        return f"<bhoga: {self.provider_id} [{self.status}] {pct} — {self.reason}>"

# ── Provider helpers ───────────────────────────────────────────────────────

def model_family(m: str) -> str:
    return next((f for p, f in _FAM.items() if m.lower().startswith(p)), "other")

def hierarchy_for(model_id: str, overrides: dict | None = None) -> list[str]:
    fam = model_family(model_id)
    return ifnone((overrides or {}).get(fam), _DEFAULTS["hierarchy"].get(fam, ["copilot"]))

def cfg(pid: str) -> dict:
    return _DEFAULTS["providers"].get(pid, {})

def blank(pid: str, mid: str) -> ProviderQuota:
    c = cfg(pid)
    return ProviderQuota(pid, mid, cadence=BillingCadence(c.get("cadence","monthly")),
                         priority=c.get("priority", 99))

def to_hermes_model(pid: str, mid: str) -> str:
    "Convert bhoga provider_id + model_id → Hermes 'provider/model' string."
    hp = _DEFAULTS["hermes_provider"].get(pid, "main")
    return f"{hp}/{mid}"

# ── Model list fetchers ────────────────────────────────────────────────────

def _fetch(url: str, headers: dict, timeout=8) -> list[dict]:
    try:
        r = httpx.get(url, headers=headers, timeout=timeout)
        return r.json().get("data") or r.json() if r.is_success else []
    except Exception: return []

def fetch_models(pid: str) -> list[str]:
    "Fetch live model list for a provider. Returns [] if creds missing or request fails."
    c = cfg(pid)
    if not (url := c.get("models_url")): return []
    key = os.getenv(c.get("auth_env",""), "")
    if not key: return []
    hdrs = ({"x-api-key": key, "anthropic-version": "2023-06-01"} if pid == "anthropic_api"
            else {"Authorization": f"Bearer {key}",
                  **({"X-GitHub-Api-Version": "2022-11-28"} if pid == "copilot" else {})})
    rows = _fetch(url, hdrs)
    return [r.get("id","") for r in rows if r.get("id")]

# ── CLI quota parse ────────────────────────────────────────────────────────

_CLAUDE_P = ('Parse Claude Code `/usage` stdout. Return ONLY JSON:\n'
             '{{"plan":str,"consumed":int,"ceiling":int,"cadence":"hourly|daily|weekly|monthly",'
             '"reset_in_seconds":int}} — unknown fields: null.\n\nOutput:\n{raw}')

_CODEX_P  = ('Parse Codex `/status` stdout. Return ONLY JSON:\n'
             '{{"plan":str,"pct_remaining":float,"cadence":"hourly|daily|weekly|monthly",'
             '"reset_at_iso":str}} — unknown fields: null.\n\nOutput:\n{raw}')

def _cli(cmd: list[str]) -> str | None:
    try: return subprocess.run(cmd, capture_output=True, text=True, timeout=10).stdout
    except (FileNotFoundError, subprocess.TimeoutExpired): return None

def _parse_json(llm_fn: Callable, prompt: str) -> dict | None:
    try: return json.loads(llm_fn(prompt))
    except Exception: return None

def parse_claude_quota(llm_fn: Callable) -> ProviderQuota | None:
    if not (raw := _cli(["claude", "/usage"])): return None
    if not (d := _parse_json(llm_fn, _CLAUDE_P.format(raw=raw))): return None
    now = datetime.now(timezone.utc)
    q = ProviderQuota(
        "claude_code", "*", plan=ifnone(d.get("plan"), "unknown"),
        ceiling=d.get("ceiling") or cfg("claude_code")["plans"].get(d.get("plan",""), None),
        cadence=BillingCadence(ifnone(d.get("cadence"), "monthly")),
        reset_at=now + timedelta(seconds=d["reset_in_seconds"]) if d.get("reset_in_seconds") else None,
        consumed=ifnone(d.get("consumed"), 0), raw_cli=raw, priority=1,
    )
    q.recompute(); return q

def parse_codex_quota(llm_fn: Callable) -> ProviderQuota | None:
    if not (raw := _cli(["codex", "/status"])): return None
    if not (d := _parse_json(llm_fn, _CODEX_P.format(raw=raw))): return None
    pct = d.get("pct_remaining")
    q = ProviderQuota(
        "codex", "*", plan=ifnone(d.get("plan"), "unknown"),
        cadence=BillingCadence(ifnone(d.get("cadence"), "weekly")),
        reset_at=datetime.fromisoformat(d["reset_at_iso"]) if d.get("reset_at_iso") else None,
        ceiling=1_000_000,    # synthetic ceiling: Codex only gives % remaining
        calibrated=int(pct * 1_000_000) if pct is not None else None,
        raw_cli=raw, priority=1,
    )
    q.recompute(); return q

# ── Header calibration ─────────────────────────────────────────────────────

def calibrate(q: ProviderQuota, headers: dict) -> None:
    "Mutate quota in-place from HTTP response headers. Idempotent."
    c, h = cfg(q.provider_id), {k.lower(): v for k, v in headers.items()}
    if not (rem_key := c.get("hdr_rem")): return
    if rem := h.get(rem_key):
        q.calibrated    = int(rem)
        q.calibrated_at = datetime.now(timezone.utc)
    if rst := h.get(c.get("hdr_rst","")):
        try: q.reset_at = datetime.fromtimestamp(int(rst), tz=timezone.utc)
        except ValueError: pass
    q.recompute()

def mark_throttled(q: ProviderQuota, retry_after: int = 60) -> None:
    q.backoff_until = datetime.now(timezone.utc) + timedelta(seconds=retry_after)
    q.status = ProviderStatus.THROTTLED

# ── State persistence ──────────────────────────────────────────────────────

def _ser(q: ProviderQuota) -> dict:
    return {k: (v.isoformat() if isinstance(v, datetime) else
                v.value      if hasattr(v, "value") else v)
            for k, v in q.__dict__.items()}

def _deser(d: dict) -> ProviderQuota:
    for f in ("reset_at","backoff_until","calibrated_at"):
        if d.get(f): d[f] = datetime.fromisoformat(d[f])
    d["cadence"] = BillingCadence(d.get("cadence","monthly"))
    d["status"]  = ProviderStatus(d.get("status","unknown"))
    return ProviderQuota(**d)

def save_state(state: dict[str, ProviderQuota]) -> None:
    STATE_PATH.parent.mkdir(parents=True, exist_ok=True)
    STATE_PATH.write_text(json.dumps({k: _ser(v) for k,v in state.items()}, indent=2))

def load_state() -> dict[str, ProviderQuota]:
    if not STATE_PATH.exists(): return {}
    try:
        qs = {k: _deser(v) for k,v in json.loads(STATE_PATH.read_text()).items()}
        for q in qs.values(): q.reset_if_due()
        return qs
    except Exception: return {}

# ── Router ─────────────────────────────────────────────────────────────────

class Router:
    """
    Zero-server provider router. One instance per harness session.

    Usage:
        router = Router(llm_fn=cheap_model)           # init once
        rec    = router.best_for("claude-opus-4")     # before every request
        router.record_turn(rec.provider_id, tokens,   # after every response
                           model_id, headers, status)
    """

    def __init__(self, llm_fn: Callable | None = None,
                 priority_override: dict | None = None,
                 low_threshold: float = 0.20):
        self._llm, self._ov, self._thr = llm_fn, priority_override or {}, low_threshold
        self._state: dict[str, ProviderQuota] = load_state()
        self._pool  = ThreadPoolExecutor(max_workers=2, thread_name_prefix="bhoga")
        self._pool.submit(self._init_bg)

    def _init_bg(self) -> None:
        for pid in _DEFAULTS["providers"]:
            for mid in fetch_models(pid):
                self._state.setdefault(f"{pid}:{mid}", blank(pid, mid))
        if self._llm:
            for fn in (parse_claude_quota, parse_codex_quota):
                if q := fn(self._llm): self._merge(q)

    def _merge(self, q: ProviderQuota) -> None:
        key = f"{q.provider_id}:{q.model_id}"
        if e := self._state.get(key):
            if q.ceiling: e.ceiling  = q.ceiling
            if q.reset_at: e.reset_at = q.reset_at
            if q.plan != "unknown": e.plan = q.plan
            e.recompute(self._thr)
        else: self._state[key] = q

    def best_for(self, model_id: str) -> RouterRecommendation:
        "Instant — reads from in-memory state. Call before every completion."
        pids       = hierarchy_for(model_id, self._ov)
        candidates = [self._state.get(f"{p}:{model_id}") or blank(p, model_id) for p in pids]
        usable     = [q for q in candidates if q.is_usable(self._thr)]

        if not usable:
            best = min(candidates, key=lambda q: q.priority)
            return self._rec(best, model_id, "all_exhausted", candidates)

        known   = [q for q in usable if q.quota_pct >= 0]
        unknown = [q for q in usable if q.quota_pct < 0]
        if known:
            s    = sorted(known, key=lambda q: (q.quota_pct, -q.priority), reverse=True)
            best = s[0]
            reason = ("priority_tiebreak" if len(s) > 1 and s[0].quota_pct == s[1].quota_pct
                      else "only_available" if len(known) == 1 else "highest_quota")
        else:
            best, reason = min(unknown, key=lambda q: q.priority), "unknown_quota"
        return self._rec(best, model_id, reason, candidates)

    def _rec(self, q, model_id, reason, candidates) -> RouterRecommendation:
        return RouterRecommendation(
            q.provider_id, model_id, q.quota_pct, q.status, reason,
            hermes_model=to_hermes_model(q.provider_id, model_id),
            all_providers={c.provider_id: c for c in candidates},
        )

    def record_turn(self, provider: str, tokens: int, model_id: str = "*",
                    headers: dict | None = None, status_code: int = 200) -> None:
        "Non-blocking. Submit after every completion — harness contract."
        self._pool.submit(self._record_bg, provider, tokens, model_id, headers, status_code)

    def _record_bg(self, provider, tokens, model_id, headers, status_code) -> None:
        key = f"{provider}:{model_id}"
        q   = self._state.setdefault(key, blank(provider, model_id))
        q.consumed    += tokens
        q.n_requests  += 1
        if status_code == 429:
            mark_throttled(q, int((headers or {}).get("retry-after", 60)))
        elif headers: calibrate(q, headers)
        q.recompute(self._thr)
        save_state(self._state)

    @property
    def state(self) -> dict[str, ProviderQuota]: return dict(self._state)
    def __repr__(self):
        ok = sum(1 for q in self._state.values() if q.status == ProviderStatus.OK)
        return f"<bhoga.Router: {ok}/{len(self._state)} OK>"

# ── Standalone CLI ─────────────────────────────────────────────────────────

_ICON = {ProviderStatus.OK:"🟢", ProviderStatus.LOW:"🟡",
         ProviderStatus.THROTTLED:"🔴", ProviderStatus.EXHAUSTED:"⛔",
         ProviderStatus.UNKNOWN:"⚪"}

def check_quota() -> None:
    "Print current quota table. Entry point: bhoga-check."
    s = load_state()
    if not s: return print("No state. Run a harness session first.")
    print(f"\n{'Provider':<20}{'Model':<40}{'St':<4}{'Status':<12}{'Quota':>7}  Reset")
    print("─" * 88)
    for q in sorted(s.values(), key=lambda q: (q.priority, q.provider_id)):
        pct = f"{q.quota_pct:.0%}" if q.quota_pct >= 0 else "  ?"
        rst = q.reset_at.strftime("%m-%d %H:%M") if q.reset_at else "—"
        print(f"{q.provider_id:<20}{q.model_id:<40}{_ICON[q.status]}  {q.status:<11}{pct:>6}  {rst}")
    print()
```

### Hermes Session Example

A Hermes skill or startup hook that selects the best provider at boot:

```python
# ~/.hermes/skills/bhoga-startup/run.py
# Hermes calls this as a skill at session start.
# Required env: ANTHROPIC_API_KEY, OPENAI_API_KEY, GITHUB_TOKEN

import subprocess
from bhoga import Router

TARGET_MODEL = "claude-opus-4-20250514"   # or read from ~/.hermes/config.yaml

# llm_fn: use Hermes' own cheap model for CLI parsing
# Hermes exposes this via the skill context — replace with your actual call
def cheap_call(prompt: str) -> str:
    import httpx
    r = httpx.post("http://localhost:1234/v1/chat/completions",
                   json={"model": "gpt-4o-mini", "messages": [{"role":"user","content":prompt}]},
                   timeout=15)
    return r.json()["choices"][0]["message"]["content"]

router = Router(llm_fn=cheap_call)
rec    = router.best_for(TARGET_MODEL)

if rec.available:
    # Apply to Hermes config for this session
    subprocess.run(["hermes", "config", "set", "model", rec.hermes_model], check=True)
    print(f"[bhoga] Routed to {rec.hermes_model} ({rec.quota_pct:.0%} quota) — {rec.reason}")
else:
    print(f"[bhoga] WARNING: all providers exhausted for {TARGET_MODEL}")
    # Optionally fall back to a different model entirely:
    # subprocess.run(["hermes", "config", "set", "model", "copilot/claude-sonnet-4"], check=True)
```

### Hermes `config.yaml` — Auxiliary Model Slots

bhoga can also route auxiliary/compression model slots.
Hermes uses the same `provider/model/base_url` pattern for all slots:

```yaml
# ~/.hermes/config.yaml — written by bhoga at session start

model: anthropic/claude-opus-4-20250514    # ← set by bhoga.best_for()

compression:
  summary_provider: codex                  # ← if openai_api has more quota
  summary_model: gpt-4o-mini

auxiliary:
  vision:
    provider: copilot                      # ← fallback if anthropic exhausted
    model: claude-opus-4-20250514
```

To write this programmatically from bhoga:

```python
import yaml
from pathlib import Path

def apply_to_hermes(router: Router, config_path=Path.home()/".hermes"/"config.yaml"):
    "Write best providers for main + compression + auxiliary vision slots."
    cfg   = yaml.safe_load(config_path.read_text()) if config_path.exists() else {}
    main  = router.best_for("claude-opus-4-20250514")
    summ  = router.best_for("gpt-4o-mini")           # compression summariser
    vis   = router.best_for("claude-opus-4-20250514") # vision

    cfg["model"] = main.hermes_model
    cfg.setdefault("compression", {}).update(
        summary_provider=_DEFAULTS["hermes_provider"].get(summ.provider_id, "main"),
        summary_model=summ.model_id,
    )
    cfg.setdefault("auxiliary", {}).setdefault("vision", {}).update(
        provider=_DEFAULTS["hermes_provider"].get(vis.provider_id, "main"),
        model=vis.model_id,
    )
    config_path.write_text(yaml.dump(cfg, default_flow_style=False))
    print(f"[bhoga] Hermes config updated: main={main.hermes_model}, "
          f"compression={summ.hermes_model}, vision={vis.hermes_model}")
```

## `record_turn()` — Why It's Required

Without `record_turn()`, bhoga cannot accumulate consumption or calibrate from
response headers. The router still works (uses CLI-parsed ceilings only) but
degrades to coarse-grained quota estimates. Always call it after every completion.

## Quota Discovery Flow

On `Router()` init (background thread):
1. `claude /usage` → LLM extracts plan ceiling + reset time
2. `codex /status` → LLM extracts % remaining + reset time
3. `/v1/models`, `/catalog/models` → live model lists per provider
4. `~/.cache/bhoga/state.json` → restore consumed counts; reset if period elapsed

On `record_turn()` (background thread, after every completion):
1. Accumulate `consumed` tokens + `n_requests`
2. Read `x-ratelimit-remaining-tokens` + reset header → calibrate
3. Write `~/.cache/bhoga/state.json`

- GitHub Copilot rate limits: https://docs.github.com/copilot/concepts/rate-limits
- Hermes configuration: https://hermes-agent.nousresearch.com/docs/user-guide/configuration

## 12. Tests — `tests/test_bhoga.py`

```python
import pytest
from datetime import datetime, timezone
from bhoga.bhoga import (ProviderQuota, RouterRecommendation, Router,
                          BillingCadence, ProviderStatus, blank)

def quota(pid, pct=0.8, priority=1) -> ProviderQuota:
    q = ProviderQuota(pid, "claude-opus-4", ceiling=1_000_000,
                      consumed=int(1_000_000 * (1 - pct)), priority=priority,
                      cadence=BillingCadence.MONTHLY)
    q.recompute(); return q

def make_router(state: dict[str, ProviderQuota]) -> Router:
    r = Router(llm_fn=None)
    r._state = state; return r

def test_highest_quota_wins():
    r = make_router({
        "claude_code:claude-opus-4":   quota("claude_code",   pct=0.3, priority=1),
        "anthropic_api:claude-opus-4": quota("anthropic_api", pct=0.8, priority=2),
    })
    rec = r.best_for("claude-opus-4")
    assert rec.provider_id == "anthropic_api"
    assert rec.reason == "highest_quota"

def test_priority_tiebreak():
    r = make_router({
        "claude_code:claude-opus-4":   quota("claude_code",   pct=0.5, priority=1),
        "anthropic_api:claude-opus-4": quota("anthropic_api", pct=0.5, priority=2),
    })
    rec = r.best_for("claude-opus-4")
    assert rec.provider_id == "claude_code"
    assert rec.reason == "priority_tiebreak"

def test_exhausted_skipped():
    r = make_router({
        "claude_code:claude-opus-4":   quota("claude_code",   pct=0.01, priority=1),
        "anthropic_api:claude-opus-4": quota("anthropic_api", pct=0.5,  priority=2),
    })
    rec = r.best_for("claude-opus-4")
    assert rec.provider_id == "anthropic_api"

def test_unknown_quota_routes_optimistically():
    r = make_router({"claude_code:*": blank("claude_code", "*")})
    rec = r.best_for("claude-opus-4")
    assert rec.available
    assert rec.reason == "unknown_quota"

def test_hermes_model_string():
    r = make_router({"anthropic_api:claude-opus-4-20250514":
                     quota("anthropic_api", pct=0.9, priority=2)})
    r._state["anthropic_api:claude-opus-4-20250514"].model_id = "claude-opus-4-20250514"
    rec = r.best_for("claude-opus-4-20250514")
    assert rec.hermes_model == "anthropic/claude-opus-4-20250514"

def test_record_turn_accumulates():
    r = make_router({"claude_code:*": blank("claude_code", "*")})
    r._record_bg("claude_code", 1500, "*", {}, 200)
    assert r._state["claude_code:*"].consumed == 1500
    assert r._state["claude_code:*"].n_requests == 1

def test_throttle_on_429():
    r = make_router({"claude_code:*": blank("claude_code", "*")})
    r._record_bg("claude_code", 0, "*", {"retry-after": "30"}, 429)
    assert r._state["claude_code:*"].status == ProviderStatus.THROTTLED
```
